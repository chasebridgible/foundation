  function ensureGtag() {
    if (typeof window.gtag === 'function') return; // tag already on the page

  function retailerForHost(host) {
    host = (host || '').toLowerCase();
    if (/(^|\.)(amazon\.|amzn\.)/.test(host)) return 'amazon';
    if (/oreilly\.com$/.test(host) || /(^|\.)oreilly\.com/.test(host)) return 'oreilly';

    return null;

    return (el && (el.textContent || '')).replace(/\s+/g, ' ').trim().slice(0, 100);

    if (!t || !t.closest) return;
    var a = t.closest('a[href]');
    if (!a) return;
